## 拓扑排序
[B3644 【模板】拓扑排序 / 家谱树](https://www.luogu.com.cn/problem/B3644)
```cpp
//洛谷P3644完整代码
#include<iostream>
#include<vector>
#include<queue>
using namespace std;

struct node{
	int enter;
	vector<int> nxt;
	int len;
}a[105];
int n;
queue<int> Q;

int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		int tmp;
		cin>>tmp;
		while(tmp!=0){
			a[i].nxt.push_back(tmp);
			a[tmp].enter++;
			a[i].len++;
			cin>>tmp;
		}
	}
	for(int i=1;i<=n;i++){
		if(a[i].enter==0) Q.push(i);
	}
	while(!Q.empty()){
		int t=Q.front();
		Q.pop();
		for(int i=0;i<a[t].len;i++){
			int tmp=a[t].nxt[i];
			a[tmp].enter--;
			if(a[tmp].enter==0) Q.push(tmp);
		}
		cout<<t<<' ';
	}
}
```
