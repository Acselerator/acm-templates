# 二分答案

[P2440 木材加工](https://www.luogu.com.cn/problem/P2440)

```cpp
//洛谷P2440 完整代码
#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=1e5+5;

ll n,k;
ll a[maxn];

bool check(ll x){
	ll ret=0;
	for(int i=1;i<=n;i++){
		ret+=a[i]/x;
	}
	if(ret<k) return 1;
	else return 0;
}

int main(){
	scanf("%lld%lld",&n,&k);
	for(int i=1;i<=n;i++){
		scanf("%lld",&a[i]);
	}
	ll l=1,r=1e8+1;
	while(l<r){
		ll mid=(l+r)>>1;
		if(check(mid)) r=mid;
		else l=mid+1;
	}
	if(l-1<0) printf("0\n");
	else printf("%lld\n",l-1);
}
```
