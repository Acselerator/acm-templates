# 线段树

此处以询问区间和为例。实际上线段树可以处理很多符合结合律的操作。

```cpp
//洛谷P3372 完整代码
#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=1e5+5;

int n,m;
ll sa[maxn];

struct segment_tree{
	ll tre[maxn<<2],laz[maxn<<2]; //tre为新树节点值，laz为懒标记
	#define ls (x<<1)
	#define rs (x<<1|1)
	#define mid ((l+r)>>1) //二分分块
	
	void update(int x){
		tre[x]=tre[ls]+tre[rs];
	}
	
	void build(int x,int l,int r){ //建树
		if(l==r){
			tre[x]=sa[l];
			return;
		}
		build(ls,l,mid);
		build(rs,mid+1,r);
		update(x);
	}
	
	void add(int x,int l,int r,int lt,int rt,ll val){
		if(l>=lt&&r<=rt){
			tre[x]+=val*(r-l+1);
			laz[x]+=val;
			return;
		}
		if(laz[x]) pushdown(x,l,r); //向下传导lazy tag
		if(lt<=mid) add(ls,l,mid,lt,rt,val);
		if(rt>mid) add(rs,mid+1,r,lt,rt,val);
		update(x); //递归回溯时向上更新
	}
	
	void pushdown(int x,int l,int r){
		tre[ls]+=laz[x]*(mid-l+1);
		tre[rs]+=laz[x]*(r-mid);
		laz[ls]+=laz[x];
		laz[rs]+=laz[x];
		laz[x]=0;
	}
	
	ll query(int x,int l,int r,int lt,int rt){
		if(l>=lt&&r<=rt){
			return tre[x];
		}
		ll res=0;
		if(laz[x]) pushdown(x,l,r); //向下传导lazy tag
		if(mid>=lt) res+=query(ls,l,mid,lt,rt);
		if(rt>mid) res+=query(rs,mid+1,r,lt,rt);
		return res;
	}
}T;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%lld",&sa[i]);
	}
	T.build(1,1,n);
	for(int i=1;i<=m;i++){
		int ch;
		scanf("%d",&ch);
		if(ch==1){
			int x,y;
			ll k;
			scanf("%d%d%lld",&x,&y,&k);
			T.add(1,1,n,x,y,k); //区间修改
		}else{
			int x,y;
			scanf("%d%d",&x,&y);
			printf("%lld\n",T.query(1,1,n,x,y)); //区间查询
		}
	}
}
```
