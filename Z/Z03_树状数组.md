# 树状数组

树状数组支持的操作：

1. 区间和、区间异或和、区间乘积和 RMQ（显然，支持的操作都具有交换律，这也算是树状数组的一大特性吧）

2. 单点修改（朴素的树状数组结构不支持区间修改，当然也可以普及成区间修改结构但我们先不提）

另：权值树状数组——tre存储值为原数组中对应下标的数出现的次数

**树状数组 1**：单点修改，区间查询

[P3374 【模板】树状数组 1](https://www.luogu.com.cn/problem/P3374)

```cpp
//洛谷P3374 完整代码 
#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=5e5+5;

int n,m;

struct tree_sequence{
	ll tre[maxn];
	int lowbit(int x){ //二进制数最低的1位 
		return x&(-x);
	}
	void add(int pos,int val){ //每加上一个lowbit位都要add 
		while(pos<=n){
			tre[pos]+=val;
			pos+=lowbit(pos);
		}
	}
	ll presum(int pos){ //每减去一个lowbit位之和  
		ll res=0;
		while(pos){
			res+=tre[pos];
			pos-=lowbit(pos);
		}
		return res;
	}
	ll query(int l,int r){
		return presum(r)-presum(l-1); //前缀和之差计算区间和 
	}
}T;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		int t;
		scanf("%d",&t);
		T.add(i,t);
	}
	for(int i=1;i<=m;i++){
		int ch,x,y;
		scanf("%d%d%d",&ch,&x,&y);
		if(ch==1){
			T.add(x,y);
		}else{
			printf("%lld\n",T.query(x,y));
		}
	}
}
```

**树状数组 2**：区间修改，单点查询

[P3368 【模板】树状数组 2](https://www.luogu.com.cn/problem/P3368)

```cpp
//洛谷P3368 完整代码 
#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=5e5+5;

int n,m;

struct tree_sequence{ //树状数组存的是差分 
	ll tre[maxn];
	int lowbit(int x){ //二进制数最低的1位 
		return x&(-x);
	}
	void add(int pos,int val){ //每加上一个lowbit位都要add 
		while(pos<=n){
			tre[pos]+=val;
			pos+=lowbit(pos);
		}
	}
	ll presum(int pos){ //每减去一个lowbit位之和  
		ll res=0;
		while(pos){
			res+=tre[pos];
			pos-=lowbit(pos);
		}
		return res;
	}
}T;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		int t;
		scanf("%d",&t);
		T.add(i,t);
		T.add(i+1,-t);
	}
	for(int i=1;i<=m;i++){
		int ch,x,y;
		scanf("%d",&ch);
		if(ch==1){
			int x,y,k;
			scanf("%d%d%d",&x,&y,&k);
			T.add(x,k);
			T.add(y+1,-k);
		}else{
			int x;
			scanf("%d",&x);
			printf("%lld\n",T.presum(x)); //单点查询即求差分前缀和 
		}
	}
}
```
