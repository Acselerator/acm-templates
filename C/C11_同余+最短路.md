# 同余+最短路

[P3403 跳楼机](https://www.luogu.com.cn/problem/P3403)

```cpp
//洛谷P3403 完整代码
#include<iostream>
#include<vector>
#include<queue>
using namespace std;
const int maxn=1e5+5;
const long long inf = (1ll<<63)-1;
typedef long long ll;

struct edge{
	int pos,val;
};
vector<edge> e[maxn];

struct node{
	ll dis;
	int pos;
	bool operator <(const node&x)const{
		return x.dis<dis;
	}
};
priority_queue<node> q;

ll h;
int x,y,z;
ll d[maxn];
bool vis[maxn];
ll ans;

void dj(int u){
	q.push((node){0,u});
	d[u]=0;
	while(!q.empty()){
		u=q.top().pos;
		q.pop();
		if(vis[u]) continue;
		vis[u]=1;
		for(int i=0;i<e[u].size();i++){
			int v=e[u][i].pos;
			int w=e[u][i].val;
			if(d[v]>d[u]+w){
				d[v]=d[u]+w;
				q.push((node){d[v],v});
			}
		}
	}
}

int main(){
	scanf("%lld",&h);
	h--; //便于取模运算
	scanf("%d%d%d",&x,&y,&z);
	for(int i=0;i<x;i++){ //根据对x同余关系，将楼层分为x类，建立x个结点
		e[i].push_back((edge){(i+y)%x,y});
		e[i].push_back((edge){(i+z)%x,z});
		d[i]=inf; //以第i个结点为起点所能到达的最低楼层
	}
	dj(0);
	for(int i=0;i<x;i++){
		if(h>=d[i]) ans+=(h-d[i])/x+1;
	}
	printf("%lld\n",ans);
}
```
