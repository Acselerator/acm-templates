# 树链剖分

dfs1()要处理几件事情：
- 标记每个点的深度dep[]
- 标记每个点的父亲fa[]
- 标记每个非叶子节点的子树大小(含它自己)
- 标记每个非叶子节点的重儿子编号son[]

dfs2()要处理几件事情
- 标记每个点的新编号
- 赋值每个点的初始值到新编号上
- 处理每个点所在链的顶端
- 处理每条链

顺序：**先处理重儿子再处理轻儿子**

//洛谷P3384 完整代码 
#include<iostream>
using namespace std;
const int maxn=1e5+5;

int read(){
	int x=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9'){
		if(ch=='-') f=-1;
		ch=getchar();
	}
	while(ch>='0'&&ch<='9'){
		x=x*10+ch-'0';
		ch=getchar();
	}
	return x*f;
}

struct edge{
	int to,nxt;
}e[maxn<<1];
int head[maxn],tot;

void add(int x,int y){
	e[++tot].to=y;
	e[tot].nxt=head[x];
	head[x]=tot;
}

int n,m,r,mod;
int cnt;
int w[maxn],wt[maxn];
int fa[maxn],dep[maxn],siz[maxn],son[maxn];
int id[maxn],top[maxn];
//-----------------------------------------------------------------以下为线段树 
struct segment_tree{ 
	int tre[maxn<<2],laz[maxn<<2]; //tre为新树节点值，laz为懒标记
	#define ls (x<<1)
	#define rs (x<<1|1)
	#define mid ((l+r)>>1) //二分分块
	
	void update(int x){
		tre[x]=tre[ls]+tre[rs];
		tre[x]%=mod;
	}
	
	void build(int x,int l,int r){ //建树
		if(l==r){
			tre[x]=wt[l];
			tre[x]%=mod;
			return;
		}
		build(ls,l,mid);
		build(rs,mid+1,r);
		update(x);
	}
	
	void add(int x,int l,int r,int lt,int rt,int val){
		if(l>=lt&&r<=rt){
			tre[x]+=val*(r-l+1);
			tre[x]%=mod;
			laz[x]+=val;
			return;
		}
		if(laz[x]) pushdown(x,l,r); //向下传导lazy tag
		if(lt<=mid) add(ls,l,mid,lt,rt,val);
		if(rt>mid) add(rs,mid+1,r,lt,rt,val);
		update(x); //递归回溯时向上更新
	}
	
	void pushdown(int x,int l,int r){
		tre[ls]+=laz[x]*(mid-l+1);
		tre[ls]%=mod;
		tre[rs]+=laz[x]*(r-mid);
		tre[rs]%=mod;
		laz[ls]+=laz[x];
		laz[rs]+=laz[x];
		laz[x]=0;
	}
	
	int query(int x,int l,int r,int lt,int rt){
		if(l>=lt&&r<=rt){
			return tre[x];
		}
		int res=0;
		if(laz[x]) pushdown(x,l,r); //向下传导lazy tag
		if(mid>=lt) res+=query(ls,l,mid,lt,rt);
		if(rt>mid) res+=query(rs,mid+1,r,lt,rt);
		return res%mod; //注意在线段树中时刻取余 
	}
}T;
//-----------------------------------------------------------------以上为线段树 
void dfs1(int x,int fath,int depth){
	fa[x]=fath;
	dep[x]=depth;
	siz[x]=1; //标记每个结点的子树大小 
	int maxson=-1; ////记录重儿子的儿子数 
	for(int i=head[x];i;i=e[i].nxt){
		int y=e[i].to;
		if(y==fath) continue;
		dfs1(y,x,depth+1);
		siz[x]+=siz[y];
		if(siz[y]>maxson){
			son[x]=y; //记录重儿子 
			maxson=siz[y];
		}
	}
}

void dfs2(int x,int topf){ //topf当前链的最顶端的节点 
	id[x]=++cnt; //每个点的新编号 
	wt[cnt]=w[x]; //把每个点的初始值赋到新编号上来 
	top[x]=topf;
	if(!son[x]) return;
	dfs2(son[x],topf); //按先处理重儿子，再处理轻儿子的顺序递归处理 
	for(int i=head[x];i;i=e[i].nxt){
		int y=e[i].to;
		if(y==fa[x]||y==son[x]) continue;
		dfs2(y,y); //对于每一个轻儿子都有一条从它自己开始的链 
	}
}

void updRange(int x,int y,int val){
	val%=mod;
	while(top[x]!=top[y]){
		if(dep[top[x]]<dep[top[y]]) swap(x,y);
		T.add(1,1,n,id[top[x]],id[x],val);
		x=fa[top[x]];
	}
	if(dep[x]>dep[y]) swap(x,y);
	T.add(1,1,n,id[x],id[y],val);
}

int qRange(int x,int y){
	int ans=0;
	while(top[x]!=top[y]){ //当两个点不在同一条链上 
		if(dep[top[x]]<dep[top[y]]) swap(x,y); //把x点改为所在链顶端的深度更深的那个点
		ans+=T.query(1,1,n,id[top[x]],id[x]); //ans加上x点到x所在链顶端这一段区间的点权和
		ans%mod;
		x=fa[top[x]]; //把x跳到x所在链顶端的那个点的上面一个点
	}  //直到两个点处于一条链上
	if(dep[x]>dep[y]) swap(x,y); //把x点改为深度更浅的那个点
	ans+=T.query(1,1,n,id[x],id[y]);
	return ans%mod;
}

void updSon(int x,int val){
	T.add(1,1,n,id[x],id[x]+siz[x]-1,val);
}

int qSon(int x){
	return T.query(1,1,n,id[x],id[x]+siz[x]-1); //子树区间右端点为id[x]+siz[x]-1 
}

int main(){
	n=read();
	m=read();
	r=read();
	mod=read();
	for(int i=1;i<=n;i++){
		w[i]=read();
	}
	for(int i=1;i<=n-1;i++){
		int x,y;
		x=read();
		y=read();
		add(x,y);
		add(y,x);
	}
	dfs1(r,0,1);
	dfs2(r,r);
	T.build(1,1,n);
	for(int i=1;i<=m;i++){
		int ch,x,y,z;
		ch=read();
		if(ch==1){
			x=read(),y=read(),z=read();
			updRange(x,y,z);
		}else if(ch==2){
			x=read(),y=read();
			printf("%d\n",qRange(x,y));
		}else if(ch==3){
			x=read(),z=read();
			updSon(x,z);
		}else{
			x=read();
			printf("%d\n",qSon(x));
		}
	}
}
