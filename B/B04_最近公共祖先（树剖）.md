# 最近公共祖先（树剖）

基本思想是将树按一定方式剖分成链，随后便可以在链上进行快速操作求得 $LCA$，单次求解的时间复杂度在 $O(logn)$。

其中求LCA的方法是让x、y跳到同一条链，返回较深者

[P3379 【模板】最近公共祖先（LCA）](https://www.luogu.com.cn/problem/P3379)

```cpp
//洛谷P3379 完整代码
#include<iostream>
using namespace std;
const int maxn=5e5+5;
const int maxm=5e5+5;

struct edge{
    int to,nxt;
}e[maxm<<1];
int head[maxn],tot;
void add(int x,int y){
    e[++tot].to=y;
    e[tot].nxt=head[x];
    head[x]=tot;
}

int n,m,s;
int siz[maxn],fa[maxn],dep[maxn]; //子树大小，父结点，深度
int son[maxn],top[maxn]; //重儿子，链顶结点

void dfs1(int x,int fath){ //求siz,fa,dep,son
    siz[x]=1;
    fa[x]=fath;
    dep[x]=dep[fath]+1;
    for(int i=head[x];i;i=e[i].nxt){
        int y=e[i].to;
        if(y==fath) continue;
        dfs1(y,x);
        siz[x]+=siz[y];
        if(siz[y]>siz[son[x]]){
            son[x]=y;
        }
    }
}
void dfs2(int x,int tp){ //求top
    top[x]=tp;
    if(!son[x]) return;
    dfs2(son[x],tp);
    for(int i=head[x];i;i=e[i].nxt){
        int y=e[i].to;
        if(y==fa[x]||y==son[x]) continue;
        dfs2(y,y);
    }
}
int LCA(int x,int y){
    while(top[x]!=top[y]){ //将x,y移到同一条链
        if(dep[top[x]]<dep[top[y]]){
            swap(x,y);
        }
        x=fa[top[x]]; //较深的点跳到链顶的父亲
    }
    return dep[x]<dep[y]?x:y; //返回较深的点作为LCA
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    cin>>n>>m>>s;
    for(int i=1;i<n;i++){
        int x,y;
        cin>>x>>y;
        add(x,y);
        add(y,x);
    }
    dfs1(s,0); //树链剖分的两次dfs
    dfs2(s,s);
    for(int i=1;i<=m;i++){
        int x,y;
        cin>>x>>y;
        cout<<LCA(x,y)<<"\n";
    }
}
```
