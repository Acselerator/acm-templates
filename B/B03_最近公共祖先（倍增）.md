# 最近公共祖先（倍增）

[P3379 【模板】最近公共祖先（LCA）](https://www.luogu.com.cn/problem/P3379)
```cpp
//洛谷P3379完整代码
#include<iostream>
using namespace std;
const int maxn=5e5+10;

struct zzz{
	int to,nxt;
}edge[maxn<<1]; //链式前向星
int head[maxn],tot;
int depth[maxn],fa[maxn][20],lg[maxn];
int n,m,s;

void add(int x,int y){
	edge[++tot].to=y;
	edge[tot].nxt=head[x];
	head[x]=tot;
}
void dfs(int now,int fath){ //now表示当前节点，fath表示它的父亲节点
	fa[now][0]=fath;
	depth[now]=depth[fath]+1;
	for(int i=1;i<=lg[depth[now]];i++){
		fa[now][i]=fa[fa[now][i-1]][i-1]; //now的2^i祖先等于now的2^(i-1)祖先的2^(i-1)祖先
	}
	for(int i=head[now];i;i=edge[i].nxt){
		if(edge[i].to!=fath) dfs(edge[i].to,now);
	}
}
int LCA(int x,int y){
	if(depth[x]<depth[y]) swap(x,y); //数学语言：不妨设x的深度 >= y的深度
	while(depth[x]>depth[y]){
		x=fa[x][lg[depth[x]-depth[y]]]; //先跳到同一深度
	}
	if(x==y) return x; //如果x是y的祖先
	for(int k=lg[depth[x]];k>=0;k--){ //不断向上跳
		if(fa[x][k]!=fa[y][k]){ //跳到LCA的下面一层
			x=fa[x][k];y=fa[y][k];
		}
	}
	return fa[x][0];
}

int main(){
	cin>>n>>m>>s;
	for(int i=1;i<=n-1;i++){
		int x,y;
		cin>>x>>y;
		add(x,y),add(y,x);
	}
	lg[1]=0;
	for(int i=2;i<=n;i++)
		lg[i]=lg[i>>1]+1; //预先算出log2(i)的值
	dfs(s,0);
	for(int i=1;i<=m;i++){
		int a,b;
		cin>>a>>b;
		cout<<LCA(a,b)<<'\n';
	}
}
```
