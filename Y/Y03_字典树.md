# 字典树 Trie

Trie树，是一种树形结构，是一种哈希树的变种。典型应用是用于统计，排序和保存大量的字符串（但不仅限于字符串）。

它的优点是：利用字符串的公共前缀来减少查询时间，最大限度地减少无谓的字符串比较，查询效率比哈希树高。

//洛谷P8306 完整代码
#include<iostream>
#include<cstring>
using namespace std;
typedef long long ll;
const int maxn=3e6+5;

struct node{
	int son[70],cnt;
}trie[maxn];
int T;
int n,q;
int num;

int getnum(char c){ //映射
	if(c>='0'&&c<='9'){
		return c-'0';
	}else if(c>='A'&&c<='Z'){
		return c-'A'+getnum('9')+1;
	}else{
		return c-'a'+getnum('Z')+1;
	}
}

void insert(string s){ //插入
	int u=0,len=s.length();
	for(int i=0;i<len;i++){
		int v=getnum(s[i]);
		if(!trie[u].son[v])
			trie[u].son[v]=++num;
		u=trie[u].son[v];
		trie[u].cnt++;
	}
}

int find(string s){ //查询
	int u=0,len=s.length();
	for(int i=0;i<len;i++){
		int v=getnum(s[i]);
		if(!trie[u].son[v])
			return 0;
		u=trie[u].son[v];
	}
	return trie[u].cnt;
} 

int main(){
	scanf("%d",&T);
	while(T--){
		for(int i=0;i<=num;i++){
			memset(trie[i].son,0,sizeof(trie[i].son));
			trie[i].cnt=0;
		}num=0;
		scanf("%d%d",&n,&q);
		for(int i=1;i<=n;i++){
			string s;
			cin>>s;
			insert(s);
		}
		for(int i=1;i<=q;i++){
			string s;
			cin>>s;
			printf("%d\n",find(s));
		}
	}
}
