# 字符串哈希

[P3370 【模板】字符串哈希](https://www.luogu.com.cn/problem/P3370)

```cpp
//洛谷P3370 完整代码
#include<iostream>
#include<set>
using namespace std;
typedef long long ll;
const int maxn=1505;

ll base=29;
ll mod=1e9+7;
set<int> se;
ll Hash[maxn]; //hash有命名冲突
int n,ans;

int main(){
	scanf("%d",&n);
	Hash[0]=0;
	for(int i=1;i<=n;i++){
		string s;
		cin>>s;
		int len=s.length();
		for(int j=0;j<len;j++){
			Hash[j+1]=(Hash[j]*base+(s[j]-'0'+1))%mod;
		}
		se.insert(Hash[len]);
	}
	printf("%d\n",se.size());
}
```
