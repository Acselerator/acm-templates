# A*
[P5507 机关](https://www.luogu.com.cn/problem/P5507)

$A^*$即带估值函数的$BFS$，这里用了优先队列，优先取出估值函数最小的方案

**估值函数：**
$$
f(n)=g(n)+h(n)
$$
- $f(n)$估值函数
- $g(n)$已走的步数
- $h(n)$对接下来将走步数的最完美估价

本题估值函数$h(x)$可以设计为：每个按钮分别旋转到1的值。由于一次操作会牵动两个按钮，因此要除以2。

```cpp
//洛谷P5507完整代码 
#include<iostream>
#include<queue>
using namespace std;
const int N=1<<24; //一个旋钮有00 01 10 11四种状态，共占24位

int g[N],fa[N],chioce[N]; //g[]存储已经走的步数
int nxt[12][4];
int ans[20]; //根据题目范围，最多走17步
int button,Start;

struct node{
	int state; //状态
	double F;
	node(int s):state(s){
		double h=0;
		F=0;
		for(int i=0;i<12;i++)
			if((s>>(i*2))&3) //不在状态1的旋钮贡献的h值
				h+=4-((s>>(i*2))&3);
		F=h/2+g[s]; //计算估值函数
	}
	bool operator <(const node&y)const{
		return F>y.F;
	}
};
priority_queue<node> q;

int main(){
	for(int i=0;i<12;i++){
		scanf("%d",&button); //第i+1个旋钮的状态
		Start|=(button-1)<<(i*2); //Start按位存储初始的所有状态，一个旋钮占两个二进制位
		for(int j=0;j<4;j++){
			scanf("%d",&nxt[i][j]);
			nxt[i][j]-=1;
		}
	}
	q.push(node(Start)); //调用构造函数，顺便算出估值函数
	g[Start]=0;
	while(!q.empty()){
		int state=q.top().state;
		q.pop();
		if(state==0) break;
		int si,sNxt,nx,nextState;
		for(int i=0;i<12;i++){
			si=(state>>(i*2))&3; //第i个旋钮的状态
			nx=nxt[i][si]; //受牵连旋钮
			sNxt=(state>>(nx*2))&3; //受牵连旋钮的状态
			nextState=state^(si<<(i*2))^(((si+1)&3)<<(i*2)); //第一个异或使对应位变零，第二个异或记录下一个状态
			nextState=nextState^(sNxt<<(nx*2))^(((sNxt+1)&3)<<(nx*2));
			if(!g[nextState]){ //如果没有访问过就可以转移新状态了
				g[nextState]=g[state]+1;
				 fa[nextState]=state; //记录操作路径
				 chioce[nextState]=i+1; //记录操作按钮
				 q.push(node(nextState));
			}
		}
	}
	int cnt=0,state=0;
	while(state!=Start){
		ans[++cnt]=chioce[state];
		state=fa[state];
	}
	printf("%d\n",cnt);
	for(int i=cnt;i;i--)
		printf("%d ",ans[i]);
} 
```
